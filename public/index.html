Frontend Integration Guide

Add this to your frontend website to connect and listen for events:

<script src="https://cdn.socket.io/4.8.1/socket.io.min.js"></script>
<script>
  const socket = io('http://localhost:1024');

  // Listen for events
  socket.on('notification', (data) => {
    console.log('Notification:', data);
  });

  socket.on('alert', (data) => {
    console.log('Alert:', data);
  });

  socket.on('emergency', (data) => {
    console.log('Emergency:', data);
  });

  socket.on('OPEN_KIND_NOTICE', (data) => {
    console.log('OPEN_KIND_NOTICE:', data);
    // Show your kind notice modal/popup here
  });

  // Or listen for any custom events
  socket.on('yourEventName', (data) => {
    console.log('Custom event:', data);
  });
</script>
